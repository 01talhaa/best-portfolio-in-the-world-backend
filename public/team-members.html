const accessToken = localStorage.getItem('accessToken');

  document.getElementById('team-container').appendChild(memberCard);

‚ûï 2. Create New Team Member

Endpoint: POST /api/v1/team-members
Headers: 
  Content-Type: application/json
  Authorization: Bearer {access_token}

Request Body:
{
  "firstName": "Jane",
  "lastName": "Smith",
  "email": "jane.smith@company.com", // optional, must be unique
  "phone": "+1-555-0124", // optional
  "position": "UI/UX Designer",
  "bio": "Creative designer with a passion for user experience...", // optional
  "profileImage": "https://example.com/profile.jpg", // optional
  "skills": ["UI Design", "UX Research", "Figma", "Adobe Creative Suite"], // optional
  "experience": 3, // optional
  "department": "Design", // optional
  "socialLinks": { // optional
    "linkedin": "https://linkedin.com/in/janesmith",
    "behance": "https://behance.net/janesmith"
  },
  "isActive": true, // optional, default: true
  "joinDate": "2024-04-01" // optional
}

Success Response (201):
{
  "success": true,
  "data": {
    "_id": "507f1f77bcf86cd799439012",
    "firstName": "Jane",
    "lastName": "Smith",
    // ... all fields
    "createdAt": "2024-04-01T00:00:00.000Z",
    "updatedAt": "2024-04-01T00:00:00.000Z"
  }
}

JavaScript Example:
const memberData = {
  firstName: 'Jane',
  lastName: 'Smith',
  position: 'UI/UX Designer',
  skills: ['UI Design', 'UX Research', 'Figma'],
  experience: 3
};

const response = await fetch('/api/v1/team-members', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
    'Authorization': `Bearer ${accessToken}`
  },
  body: JSON.stringify(memberData)
});
const result = await response.json();

{

const updates = {
  position: 'Senior Developer',
  experience: 6,
  skills: ['JavaScript', 'React', 'Node.js', 'TypeScript']

const response = await fetch(`/api/v1/team-members/${memberId}`, {

  },
  body: JSON.stringify(updates)

üóëÔ∏è 5. Delete Team Member

Endpoint: DELETE /api/v1/team-members/:id
Headers: Authorization: Bearer {access_token}

Success Response (200):
{
  "success": true,
  "data": {},
  "message": "Team member removed successfully"
}

JavaScript Example:
const response = await fetch(`/api/v1/team-members/${memberId}`, {
  method: 'DELETE',
  headers: {
    'Authorization': `Bearer ${accessToken}`
  }
});

if (response.ok) {
  // Remove from UI
  document.getElementById(`member-${memberId}`).remove();
}

üì∏ 6. Profile Image Upload

Upload profile images using the upload endpoint:

Endpoint: POST /api/v1/upload/single
Headers: Authorization: Bearer {access_token}
Body: FormData

FormData fields:
- file: (image file)
- entityType: "team-member"
- entityId: "507f1f77bcf86cd799439011"
- imageType: "profile" (automatically sets profileImage field)

JavaScript Example:
const formData = new FormData();
formData.append('file', profileImageFile);
formData.append('entityType', 'team-member');
formData.append('entityId', memberId);
formData.append('imageType', 'profile');

const response = await fetch('/api/v1/upload/single', {
  method: 'POST',
  headers: {
    'Authorization': `Bearer ${accessToken}`
  },
  body: formData
});

// Profile image will be automatically updated in team member record

üîß Field Reference

Required Fields:
- firstName: string
- lastName: string  
- position: string

Optional Fields:
- email: string (unique, valid email format)
- phone: string (valid phone number format)
- bio: string (max 1000 characters)
- profileImage: string (URL)
- skills: array of strings
- experience: number (years)
- department: string
- socialLinks: object { linkedin, github, portfolio, twitter, behance, etc. }
- isActive: boolean (default: true)
- joinDate: Date

Validation Rules:
- Email must be unique if provided
- Phone number must be in valid format
- Profile image must be valid URL
- Social links must be valid URLs
- Experience must be non-negative number

Common Use Cases:
1. Team page display: Get all active members
2. Member profiles: Get individual member details  
3. Admin panel: CRUD operations for HR management
4. Project assignments: Link members to projects
5. Skills filtering: Filter by skills or department
